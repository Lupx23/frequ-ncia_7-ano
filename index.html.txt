<script>
const url = "https://script.google.com/macros/s/AKfycbw18qBsO_MB-V15_szleAIPa2pmpZ9VvPmjPoI-5-4BVpUcoQQigGc_NRumLCFPdbGG/exec";

    body:new URLSearchParams({

      tipo:tipo
    })
  })
  .then(res=>res.text())

    document.getElementById("msg").innerText = texto;

    document.getElementById("msg").innerText="Erro ao registrar";
  });
